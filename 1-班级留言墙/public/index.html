// 替换原uploadPhoto函数 async function uploadPhoto() { const albumId = document.getElementById('albumSelect').value; const fileInput = document.getElementById('uploadFile'); // 前端提前校验 if (!albumId) { log('请选择专辑', 'error'); return; } if (!fileInput.files.length) { log('请选择照片文件', 'error'); return; } const file = fileInput.files[0]; const maxSize = 5 * 1024 * 1024; // 5MB if (file.size > maxSize) { log(`文件大小超过5MB限制（当前${(file.size/1024/1024).toFixed(2)}MB）`, 'error'); return; } // 校验文件类型 const allowedTypes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp']; if (!allowedTypes.includes(file.type)) { log('仅支持JPG/PNG/GIF/WEBP格式', 'error'); return; } const formData = new FormData(); formData.append('photo', file); formData.append('albumId', albumId); log('正在上传照片...', 'info'); try { const res = await fetch(`${API_BASE}/api/photos`, { method: 'POST', body: formData, signal: AbortSignal.timeout(30000) }); const data = await res.json(); if (data.code === 200) { log('上传成功', 'success'); await loadAlbums(); await loadPhotos(albumId); // 重置上传区 document.getElementById('uploadContent').classList.remove('show'); fileInput.value = ''; document.getElementById('fileText').innerText = '📁 选择照片文件'; } else { log(`上传失败：${data.msg}`, 'error'); } } catch (err) { log(`上传失败：${err.message}`, 'error'); } }
